# Format: HH:MM shell_command  (simple)  atau  cron_expression command
#
# Contoh di bawah di-comment (#). Buka WebUI → toggle ON untuk aktifkan.
#
# ─── Mati/hidupkan data otomatis ───
# 22:30 svc data disable
# 07:00 svc data enable
#
# ─── Mode pesawat ───
# 23:00 cmd connectivity airplane-mode enable
# 06:00 cmd connectivity airplane-mode disable
#
# ─── WiFi ───
# 23:30 svc wifi disable
# 06:30 svc wifi enable
#
# ─── Cron: matikan data setiap 2 jam (lebiih fleksibel) ───
# 0 */2 * * * svc data disable
#
# ─── Cron: start app tiap Senin/Rabu/Jumat jam 9 pagi ───
# 0 9 * * 1,3,5 am start -n com.example/.MainActivity
#
# Awali dengan # untuk nonaktifkan sementara.
#
# Data directory: /data/adb/dailyjobs/
# Log:            /data/adb/dailyjobs/run.log
# Scheduler PID:  /data/adb/dailyjobs/scheduler.pid
